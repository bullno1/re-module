532 void
533 remodule_reload(remodule_t* mod) {
534  mod->info.entry(REMODULE_OP_BEFORE_RELOAD, mod->userdata);
535 
536  // Store all static vars in a host-allocated buffer
537  int num_vars = 0;
538  size_t val_buffer_size = 0;
539  size_t name_buffer_size = 0;
540  for (
541  const remodule_var_info_t* const* itr = mod->info.var_info_begin;
542  itr != mod->info.var_info_end;
543  ++itr
544  ) {
545  if (*itr == NULL) { continue; }
546  remodule_var_info_t var_info = **itr;
547 
548  ++num_vars;
549  val_buffer_size += var_info.value_size;
550  name_buffer_size += var_info.name_length;
551  }
552 
553  void* tmp_buf = malloc(
554  num_vars * sizeof(remodule_tmp_var_storage_t)
555  + val_buffer_size
556  + name_buffer_size
557  );
558  remodule_tmp_var_storage_t* entry_ptr = tmp_buf;
559  char* data_ptr = (char*)(entry_ptr + num_vars);
560 
561  for (
562  const remodule_var_info_t* const* itr = mod->info.var_info_begin;
563  itr != mod->info.var_info_end;
564  ++itr
565  ) {
566  if (*itr == NULL) { continue; }
567  remodule_var_info_t var_info = **itr;
568 
569  remodule_tmp_var_storage_t* entry = entry_ptr++;
570 
571  entry->name = data_ptr;
572  entry->name_length = var_info.name_length;
573  data_ptr += var_info.name_length;
574 
575  entry->value = data_ptr;
576  entry->value_size = var_info.value_size;
577  data_ptr += var_info.value_size;
578 
579  memcpy(entry->name, var_info.name, var_info.name_length);
580  memcpy(entry->value, var_info.value_addr, var_info.value_size);
581  }
582 
583  remodule_dynlib_close(mod->lib);
584  mod->lib = remodule_dynlib_open(mod->path);
585  REMODULE_ASSERT(mod->lib != NULL, "Failed to reload");
586 
587  remodule_plugin_info_t* info = remodule_dynlib_find(mod->lib, REMODULE_INFO_SYMBOL_STR);
588  REMODULE_ASSERT(info != NULL, "Module does not export info struct");
589  mod->info = *info;
590 
591  // Copy vars back in
592  remodule_tmp_var_storage_t* tmp_storage = (remodule_tmp_var_storage_t*)tmp_buf;
593  for (
594  const remodule_var_info_t* const* var_itr = mod->info.var_info_begin;
595  var_itr != mod->info.var_info_end;
596  ++var_itr
597  ) {
598  if (*var_itr == NULL) { continue; }
599  remodule_var_info_t var_info = **var_itr;
600 
601  for (
602  int storage_index = 0; storage_index < num_vars; ++storage_index
603  ) {
604  remodule_tmp_var_storage_t* storage = &tmp_storage[storage_index];
605  if (
606  storage->name_length == var_info.name_length
607  && storage->value_size == var_info.value_size
608  && memcmp(storage->name, var_info.name, storage->name_length) == 0
609  ) {
610  memcpy(var_info.value_addr, storage->value, storage->value_size);
611  break;
612  }
613  }
614  }
615  free(tmp_buf);
616 
617  mod->info.entry(REMODULE_OP_AFTER_RELOAD, mod->userdata);
618 }
